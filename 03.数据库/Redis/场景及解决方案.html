模糊查询 ​

方式一：keys（不推荐）

sh
KEYS PATTERN
1
sh
redis 127.0.0.1:6379> keys xxx*
1

优点

  • 一次性返回全部key。

缺点

  • redis是单线程，大数据量查询，会阻塞其他服务，非常危险，虽然在100万个key的数据库里面执行一次需要40ms，但是在大数据库中使用仍有性能问题。

方式二：scan（推荐）

sh
SCAN key cursor [MATCH pattern] [COUNT count]
1

需要Redis2.8版本以上。同样是O（N）复杂度的scan命令，支持通配查找，scan命令或者其他的scan如SSCAN ，HSCAN，ZSCAN命令，可以不用阻塞主线程，分批遍历，并支持游标按批次迭代返回数据，所以是比较理想的选择。keys相比scan命令优点是，keys是一次返回，而scan是需要迭代多次返回。当SCAN命令的游标参数（即cursor）被设置为 0 时， 服务器将开始一次新的迭代， 而当服务器向用户返回值为 0 的游标时， 表示迭代已结束。

缺点

  • 它返回的结果有可能重复，因此需要客户端去重。
  • 消耗的总时间比keys要长。

注意事项

  • 游标必须正确传递：每次遍历需使用上次返回的游标，否则会导致重复遍历或漏遍历；
  • 避免过度依赖 MATCH：MATCH 是在遍历结果中筛选，而非提前过滤，若符合条件的键极少，会导致多次空遍历，建议结合业务场景优化匹配模式；
  • count 参数按需调整：数据量小时用默认 count=10 即可，海量数据时可增大 count（如 1000~10000），但不宜过大（避免单次操作耗时过长）；
  • 业务层去重：因 Scan 可能返回重复元素，需在业务层通过 Set 或哈希表去重；
  • 不建议在主库高频使用：虽然 Scan 非阻塞，但频繁遍历仍会占用 CPU 资源，建议在从库执行（需确保主从数据同步及时）

批量删除 ​

方式一：del（单次不超过1000个，分批次删除）

  • 主要风险：阻塞 Redis 单线程，影响其他操作
  • 影响程度：与删除的 key 数量、key 大小、Redis 负载相关
  • 关键阈值：单次删除超过 1000 个 key 就需要考虑分批次
java
@Service
public class SafeBatchDeleteService {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    /**
     * 安全的分批次删除
     */
    public long safeBatchDelete(Set<String> allKeys, int batchSize) {
        long totalDeleted = 0;
        List<String> keyList = new ArrayList<>(allKeys);
        
        for (int i = 0; i < keyList.size(); i += batchSize) {
            int end = Math.min(i + batchSize, keyList.size());
            List<String> batch = keyList.subList(i, end);
            
            Long deleted = redisTemplate.delete(batch);
            totalDeleted += deleted != null ? deleted : 0;
            
            // 添加延迟，减少对Redis的冲击
            if (i + batchSize < keyList.size()) {
                try {
                    Thread.sleep(10); // 10ms延迟
                } catch (InterruptedException e) {
                    Thread.currentThread().interrupt();
                    break;
                }
            }
            
            // 每删除1000个key打印进度
            if (totalDeleted % 1000 == 0) {
                System.out.println("已删除: " + totalDeleted + " 个key");
            }
        }
        
        return totalDeleted;
    }
}

方式二：UNLINK：Redis 4.0

java
@Service
public class UnlinkService {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    /**
     * 使用UNLINK替代DEL - 非阻塞删除
     */
    public long unlinkKeys(Set<String> keys) {
        return redisTemplate.execute((RedisCallback<Long>) connection -> {
            // UNLINK是异步非阻塞的
            byte[][] keyBytes = keys.stream()
                .map(String::getBytes)
                .toArray(byte[][]::new);
            
            // 需要根据Redis版本支持情况使用
            return connection.unlink(keyBytes);
        });
    }
    
    /**
     * 兼容性检查
     */
    public boolean supportUnlink() {
        try {
            // 测试UNLINK命令是否可用
            redisTemplate.execute((RedisCallback<Object>) connection -> {
                connection.unlink("test_key".getBytes());
                return null;
            });
            return true;
        } catch (Exception e) {
            return false;
        }
    }
}

方式三：异步删除

java
@Service
public class AsyncDeleteService {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    @Async("taskExecutor")
    public CompletableFuture<Long> asyncBatchDelete(Set<String> keys, int batchSize) {
        return CompletableFuture.supplyAsync(() -> {
            long startTime = System.currentTimeMillis();
            long deletedCount = safeBatchDelete(keys, batchSize);
            long endTime = System.currentTimeMillis();
            
            System.out.println("异步删除完成，耗时: " + (endTime - startTime) + "ms");
            return deletedCount;
        });
    }
    
    /**
     * 异步删除配置
     */
    @Configuration
    @EnableAsync
    public static class AsyncConfig {
        @Bean("taskExecutor")
        public TaskExecutor taskExecutor() {
            ThreadPoolTaskExecutor executor = new ThreadPoolTaskExecutor();
            executor.setCorePoolSize(5);
            executor.setMaxPoolSize(10);
            executor.setQueueCapacity(100);
            executor.setThreadNamePrefix("RedisDelete-");
            executor.initialize();
            return executor;
        }
    }
}

方式四：使用 Pipeline（高性能方案）

通过Pipeline模式批量删除，需注意脚本数组上限（如20万键需分多次执行）

java
@Service
public class PipelineDeleteService {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    /**
     * 使用Pipeline批量删除 - 减少网络往返
     */
    public long pipelineBatchDelete(Set<String> keys, int pipelineSize) {
        return redisTemplate.execute(new RedisCallback<Long>() {
            @Override
            public Long doInRedis(RedisConnection connection) throws DataAccessException {
                long deletedCount = 0;
                int count = 0;
                
                connection.openPipeline();
                
                for (String key : keys) {
                    connection.del(key.getBytes());
                    count++;
                    
                    // 每 pipelineSize 个命令执行一次
                    if (count % pipelineSize == 0) {
                        List<Object> results = connection.closePipeline();
                        deletedCount += pipelineSize;
                        connection.openPipeline();
                    }
                }
                
                // 执行剩余的命令
                List<Object> lastResults = connection.closePipeline();
                deletedCount += count % pipelineSize;
                
                return deletedCount;
            }
        });
    }
}

生产级综合方案 ​

java
@Service
@Slf4j
public class ProductionDeleteService {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    @Autowired
    private DeleteMonitor deleteMonitor;
    
    /**
     * 生产环境安全的批量删除
     */
    public BatchDeleteResult safeProductionDelete(String pattern, DeleteConfig config) {
        long totalDeleted = 0;
        int batchCount = 0;
        long startTime = System.currentTimeMillis();
        
        try {
            Set<String> keys;
            String cursor = "0";
            RedisConnectionFactory factory = redisTemplate.getConnectionFactory();
            
            try (RedisConnection connection = factory.getConnection()) {
                do {
                    // 使用SCAN迭代
                    Cursor<byte[]> scanCursor = connection.scan(
                        ScanOptions.scanOptions()
                            .match(pattern)
                            .count(config.getScanCount())
                            .build()
                    );
                    
                    Set<String> batchKeys = new HashSet<>();
                    while (scanCursor.hasNext()) {
                        batchKeys.add(new String(scanCursor.next()));
                        if (batchKeys.size() >= config.getBatchSize()) {
                            break;
                        }
                    }
                    
                    if (!batchKeys.isEmpty()) {
                        // 根据配置选择删除方式
                        long batchStart = System.currentTimeMillis();
                        long deleted = config.isUseUnlink() ? 
                            unlinkKeys(batchKeys) : deleteKeys(batchKeys);
                        long batchDuration = System.currentTimeMillis() - batchStart;
                        
                        totalDeleted += deleted;
                        batchCount++;
                        
                        // 记录监控
                        deleteMonitor.recordDelete(batchKeys.size(), batchDuration);
                        
                        // 进度控制
                        if (config.getDelayBetweenBatches() > 0) {
                            Thread.sleep(config.getDelayBetweenBatches());
                        }
                        
                        // 检查是否超时
                        if (System.currentTimeMillis() - startTime > config.getMaxTotalTime()) {
                            log.warn("达到最大执行时间限制，提前结束");
                            break;
                        }
                    }
                    
                    keys = batchKeys;
                    
                } while (!keys.isEmpty());
            }
            
            long totalDuration = System.currentTimeMillis() - startTime;
            
            log.info("删除完成: 总数={}, 批次={}, 耗时={}ms", 
                     totalDeleted, batchCount, totalDuration);
                     
            return BatchDeleteResult.success(totalDeleted, batchCount, totalDuration);
            
        } catch (Exception e) {
            log.error("批量删除失败", e);
            return BatchDeleteResult.error(totalDeleted, e.getMessage());
        }
    }
    
    @Data
    public static class DeleteConfig {
        private int batchSize = 500;           // 每批次大小
        private int scanCount = 1000;          // SCAN每次数量
        private int delayBetweenBatches = 10;  // 批次间延迟(ms)
        private long maxTotalTime = 300000;    // 最大总执行时间(ms)
        private boolean useUnlink = true;      // 是否使用UNLINK
    }
}

@Component
public class DeleteMonitor {
    
    private final MeterRegistry meterRegistry;
    private final Counter deleteCounter;
    private final Timer deleteTimer;
    private final DistributionSummary deleteBatchSize;
    
    public DeleteMonitor(MeterRegistry meterRegistry) {
        this.meterRegistry = meterRegistry;
        this.deleteCounter = Counter.builder("redis.delete.total")
            .register(meterRegistry);
        this.deleteTimer = Timer.builder("redis.delete.duration")
            .register(meterRegistry);
        this.deleteBatchSize = DistributionSummary.builder("redis.delete.batch.size")
            .register(meterRegistry);
    }
    
    public void recordDelete(int batchSize, long duration) {
        deleteCounter.increment(batchSize);
        deleteTimer.record(duration, TimeUnit.MILLISECONDS);
        deleteBatchSize.record(batchSize);
        
        // 告警逻辑
        if (duration > 1000) { // 删除操作超过1秒
            System.err.println("警告: 删除操作耗时过长: " + duration + "ms");
        }
    }
}
