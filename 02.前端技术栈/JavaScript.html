获取当前日期 ​

获取当前日期，如果使用today.toISOString()方法，则日期少8个时区，加上8即可

javascript
const today = new Date();
var hour = today.getHours() + 8;
today.setHours(hour);
let beginTime = util.formatDate("y-m-d", today.toISOString());

格式化金额

js
function formatNumber(num) {
  return parseFloat(parseFloat(num).toFixed(2));
}

console.log(formatNumber(1.2)); // 输出 1.2
console.log(formatNumber(1.20)); // 输出 1.2
console.log(formatNumber(1.00)); // 输出 1
console.log(formatNumber(1.19999)); // 输出 1.2
